async function copy(text) { let content = text; // اگر دکمه "همه کانفیگ‌ها" زده شد، فایل واقعی را از مخزن بخوان if(text === 'link_all') { try { const response = await fetch('configs.txt'); if (!response.ok) throw new Error(); content = await response.text(); } catch (err) { alert("کانفیگ‌ها هنوز در حال بارگذاری هستند... لحظاتی دیگر تلاش کنید."); return; } } // کپی کردن متن نهایی navigator.clipboard.writeText(content).then(() => { alert("با موفقیت کپی شد! ✅"); }).catch(err => { alert("خطا در کپی! ❌"); }); } // نمایش تاریخ آپدیت document.getElementById('lastUpdate').innerText = "آخرین آپدیت پنل: " + new Date().toLocaleString('fa-IR');
